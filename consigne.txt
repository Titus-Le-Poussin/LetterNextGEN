📋 RÉSUMÉ DU PROJET
================================================================================

Créer un portfolio personnel avec un générateur de lettres de motivation
alimenté par l'IA Claude (Anthropic API).

L'outil permet de :
- Coller une offre d'emploi
- Utiliser un prompt personnalisable pour analyser l'offre
- Remplir automatiquement les champs d'une lettre de motivation
- Générer un PDF professionnel
- Sauvegarder et réutiliser ses propres prompts et modèles de lettres

┌─────────────────────────────────────────────────────────────────────────┐
│ ÉTAPE 2.4 : Configurer les politiques RLS (Row Level Security)         │
└─────────────────────────────────────────────────────────────────────────┘

Pour chaque table, créer ces politiques :

1. Dans "Table Editor", sélectionner la table "prompts"
2. Aller dans l'onglet "RLS Policies"
3. Cliquer sur "New Policy"

   POLITIQUE 1 : Lecture publique des prompts par défaut
   ────────────────────────────────────────────────────
   - Policy name : "Allow public read for default prompts"
   - Policy command : SELECT
   - Using expression : is_default = true

   POLITIQUE 2 : Lecture des prompts de l'utilisateur
   ───────────────────────────────────────────────────
   - Policy name : "Allow users to read their own prompts"
   - Policy command : SELECT
   - Using expression : auth.uid() = user_id

   POLITIQUE 3 : Création de prompts
   ──────────────────────────────────
   - Policy name : "Allow users to create prompts"
   - Policy command : INSERT
   - With check : auth.uid() = user_id

   POLITIQUE 4 : Modification de prompts
   ──────────────────────────────────────
   - Policy name : "Allow users to update their own prompts"
   - Policy command : UPDATE
   - Using expression : auth.uid() = user_id

   POLITIQUE 5 : Suppression de prompts
   ─────────────────────────────────────
   - Policy name : "Allow users to delete their own prompts"
   - Policy command : DELETE
   - Using expression : auth.uid() = user_id

⚠️ RÉPÉTER CES 5 POLITIQUES pour la table "letter_templates"

⚠️ Pour "generated_letters", créer uniquement les politiques 2, 3, 4, 5

// Gestion de l'authentification

// Inscription
async function signUp(email, password) {
    const { data, error } = await supabaseClient.auth.signUp({
        email: email,
        password: password,
    });
    
    if (error) {
        console.error('Erreur inscription:', error.message);
        return { success: false, error: error.message };
    }
    
    return { success: true, data };
}

// Connexion
async function signIn(email, password) {
    const { data, error } = await supabaseClient.auth.signInWithPassword({
        email: email,
        password: password,
    });
    
    if (error) {
        console.error('Erreur connexion:', error.message);
        return { success: false, error: error.message };
    }
    
    return { success: true, data };
}

// Déconnexion
async function signOut() {
    const { error } = await supabaseClient.auth.signOut();
    
    if (error) {
        console.error('Erreur déconnexion:', error.message);
    }
    
    window.location.reload();
}

// Vérifier si l'utilisateur est connecté
async function checkAuth() {
    const { data: { session } } = await supabaseClient.auth.getSession();
    return session;
}

// Écouter les changements d'authentification
supabaseClient.auth.onAuthStateChange((event, session) => {
    console.log('Auth event:', event);
    updateUIBasedOnAuth(session);
});

// Mettre à jour l'interface selon l'état de connexion
function updateUIBasedOnAuth(session) {
    const authSection = document.getElementById('authSection');
    const userSection = document.getElementById('userSection');
    
    if (session) {
        // Utilisateur connecté
        authSection.style.display = 'none';
        userSection.style.display = 'block';
        document.getElementById('userEmail').textContent = session.user.email;
    } else {
        // Utilisateur non connecté
        authSection.style.display = 'block';
        userSection.style.display = 'none';
    }
}
```

// Gestion des prompts personnalisés

// Charger tous les prompts (défaut + utilisateur)
async function loadPrompts() {
    const session = await checkAuth();
    
    // Charger les prompts par défaut
    let { data: defaultPrompts } = await supabaseClient
        .from('prompts')
        .select('*')
        .eq('is_default', true);
    
    // Charger les prompts de l'utilisateur si connecté
    let userPrompts = [];
    if (session) {
        const result = await supabaseClient
            .from('prompts')
            .select('*')
            .eq('user_id', session.user.id);
        userPrompts = result.data || [];
    }
    
    return [...defaultPrompts, ...userPrompts];
}

// Sauvegarder un nouveau prompt
async function savePrompt(name, content) {
    const session = await checkAuth();
    
    if (!session) {
        alert('Vous devez être connecté pour sauvegarder un prompt');
        return { success: false };
    }
    
    const { data, error } = await supabaseClient
        .from('prompts')
        .insert({
            user_id: session.user.id,
            name: name,
            content: content,
            is_default: false
        })
        .select();
    
    if (error) {
        console.error('Erreur sauvegarde prompt:', error);
        return { success: false, error };
    }
    
    return { success: true, data };
}

// Modifier un prompt existant
async function updatePrompt(promptId, name, content) {
    const { data, error } = await supabaseClient
        .from('prompts')
        .update({ name, content, updated_at: new Date() })
        .eq('id', promptId)
        .select();
    
    if (error) {
        console.error('Erreur modification prompt:', error);
        return { success: false, error };
    }
    
    return { success: true, data };
}

// Supprimer un prompt
async function deletePrompt(promptId) {
    const { error } = await supabaseClient
        .from('prompts')
        .delete()
        .eq('id', promptId);
    
    if (error) {
        console.error('Erreur suppression prompt:', error);
        return { success: false, error };
    }
    
    return { success: true };
}

// Charger les prompts dans le select
async function populatePromptSelect() {
    const prompts = await loadPrompts();
    const select = document.getElementById('promptSelect');
    
    select.innerHTML = '';
    
    prompts.forEach(prompt => {
        const option = document.createElement('option');
        option.value = prompt.id;
        option.textContent = prompt.name + (prompt.is_default ? ' (Défaut)' : '');
        select.appendChild(option);
    });
    
    // Charger le contenu du prompt sélectionné
    if (prompts.length > 0) {
        loadPromptContent(prompts[0].id);
    }
}

// Charger le contenu d'un prompt spécifique
async function loadPromptContent(promptId) {
    const { data } = await supabaseClient
        .from('prompts')
        .select('content')
        .eq('id', promptId)
        .single();
    
    if (data) {
        document.getElementById('aiPrompt').value = data.content;
    }
}
```

└─────────────────────────────────────────────────────────────────────────┘

1. Installer Node.js si pas déjà fait (https://nodejs.org)

2. Ouvrir un terminal et installer Supabase CLI :
   
   Windows (PowerShell) :
   ```
   npm install -g supabase
   ```
   
   Mac/Linux :
   ```
   npm install -g supabase
   ```

3. Vérifier l'installation :
   ```
   supabase --version
   ```


┌─────────────────────────────────────────────────────────────────────────┐
│ ÉTAPE 5.2 : Initialiser Supabase dans le projet                        │
└─────────────────────────────────────────────────────────────────────────┘

1. Dans le terminal, aller dans ton dossier projet :
   ```
   cd chemin/vers/portfolio-projet
   ```

2. Initialiser Supabase :
   ```
   supabase init
   ```

3. Se connecter à ton projet Supabase :
   ```
   supabase link --project-ref TON_PROJECT_REF
   ```
   
   Le PROJECT_REF se trouve dans l'URL de ton projet Supabase :
   https://[PROJECT_REF].supabase.co


┌─────────────────────────────────────────────────────────────────────────┐
│ ÉTAPE 5.3 : Créer l'Edge Function                                      │
└─────────────────────────────────────────────────────────────────────────┘

1. Créer la fonction :
   ```
   supabase functions new generate-letter
   ```

2. Cela crée : supabase/functions/generate-letter/index.ts

3. Remplacer le contenu par ce code :

```typescript
import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import "https://deno.land/x/xhr@0.1.0/mod.ts";

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

serve(async (req) => {
  // Gestion des requêtes OPTIONS (CORS)
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    // Récupérer les données de la requête
    const { jobOffer, prompt } = await req.json();
    
    // Récupérer la clé API Claude depuis les secrets Supabase
    const ANTHROPIC_API_KEY = Deno.env.get('ANTHROPIC_API_KEY');
    
    if (!ANTHROPIC_API_KEY) {
      throw new Error('API key non configurée');
    }
    
    // Appeler l'API Claude
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': ANTHROPIC_API_KEY,
        'anthropic-version': '2023-06-01'
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        messages: [{
          role: 'user',
          content: `${prompt}

Voici l'offre d'emploi à analyser :

${jobOffer}`
        }]
      })
    });
    
    const data = await response.json();
    
    if (!response.ok) {
      throw new Error(data.error?.message || 'Erreur API Claude');
    }
    
    // Extraire le contenu de la réponse
    const content = data.content[0].text;
    
    // Essayer de parser le JSON
    let parsedData;
    try {
      // Nettoyer le texte (enlever les backticks si présents)
      const cleanContent = content.replace(/```json|```/g, '').trim();
      parsedData = JSON.parse(cleanContent);
    } catch (e) {
      console.error('Erreur parsing JSON:', e);
      throw new Error('Le format de réponse de Claude n\'est pas valide');
    }
    
    // Retourner les données
    return new Response(
      JSON.stringify({ success: true, data: parsedData }),
      { 
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
        status: 200 
      }
    );
    
  } catch (error) {
    console.error('Erreur:', error);
    
    return new Response(
      JSON.stringify({ 
        success: false, 
        error: error.message 
      }),
      { 
        headers: { ...corsHeaders, 'Content-Type': 'application/json' },
        status: 500 
      }
    );
  }
});
```

Créer le fichier : js/letter-generator.js

```javascript
// Appeler l'Edge Function pour générer la lettre

async function generateLetter() {
    // Récupérer les valeurs
    const jobOffer = document.getElementById('jobOffer').value;
    const prompt = document.getElementById('aiPrompt').value;
    
    if (!jobOffer) {
        alert('Veuillez coller une offre d\'emploi');
        return;
    }
    
    // Afficher le loading
    document.getElementById('loadingMessage').style.display = 'block';
    document.getElementById('errorMessage').style.display = 'none';
    document.getElementById('generateBtn').disabled = true;
    
    try {
        // Appeler l'Edge Function
        const { data, error } = await supabaseClient.functions.invoke('generate-letter', {
            body: { 
                jobOffer: jobOffer,
                prompt: prompt
            }
        });
        
        if (error) throw error;
        
        if (data.success) {
            // Remplir les champs avec les données extraites
            const extractedData = data.data;
            
            document.getElementById('field1').value = extractedData.field1 || '';
            document.getElementById('field2').value = extractedData.field2 || '';
            document.getElementById('field3').value = extractedData.field3 || '';
            document.getElementById('field4').value = extractedData.field4 || '';
            document.getElementById('field5').value = extractedData.field5 || '';
            document.getElementById('field6').value = extractedData.field6 || '';
            document.getElementById('field7').value = extractedData.field7 || '';
            
            // Mettre à jour l'aperçu
            updateAllPreviews();
            
            // Message de succès
            alert('Champs remplis automatiquement avec succès !');
        } else {
            throw new Error(data.error || 'Erreur inconnue');
        }
        
    } catch (error) {
        console.error('Erreur:', error);
        document.getElementById('errorMessage').textContent = 
            '❌ Erreur : ' + error.message;
        document.getElementById('errorMessage').style.display = 'block';
    } finally {
        document.getElementById('loadingMessage').style.display = 'none';
        document.getElementById('generateBtn').disabled = false;
    }
}
```

┌─────────────────────────────────────────────────────────────────────────┐
│ ÉTAPE 6.2 : Débugger les erreurs courantes                             │
└─────────────────────────────────────────────────────────────────────────┘

ERREUR : "Supabase is not defined"
└─> Vérifier que le script Supabase est bien chargé en premier

ERREUR : "CORS policy error"
└─> Vérifier les corsHeaders dans l'Edge Function
└─> Vérifier les Redirect URLs dans Supabase Auth

ERREUR : "API key non configurée"
└─> Vérifier que le secret est bien défini : supabase secrets list

ERREUR : "Invalid JSON response"
└─> Claude peut renvoyer du texte avant/après le JSON
└─> Le code de nettoyage dans l'Edge Function doit gérer ça

ERREUR : "Permission denied" sur les tables
└─> Vérifier les politiques RLS
└─> Vérifier que l'utilisateur est bien connecté
